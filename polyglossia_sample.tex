\section{Բաժին առաջին}
\(\zeta_H(x;p)\)-ն Հուրվիցի զետա ֆունկցիան է, որը սահմանվում է հետևյալ կերպ \cite{book:elizalde_zeta}`
%
\begin{equation}
\zeta_H(x;p)=\sum_{n=0}^{\infty}\frac{1}{(n+p)^x} \quad 0<p<1 \\
\end{equation}
%
%
\section{Բաժին երկրորդ}
